CREATE TABLE Teams (
    Team_ID INT UNSIGNED NOT NULL AUTO_INCREMENT,
    TeamName VARCHAR(255) NOT NULL,
    Established_year DATE NOT NULL,
    City VARCHAR(40) NOT NULL,
    PRIMARY KEY (TEAM_ID)
);
CREATE TABLE Players (
    Player_ID INT UNSIGNED NOT NULL AUTO_INCREMENT,
    First_Name VARCHAR(255) NOT NULL,
    Last_Name VARCHAR(255) NOT NULL,
    Team_ID INT UNSIGNED NOT NULL REFERENCES Teams(TeamID)
    PRIMARY KEY (Player_ID)
);
CREATE TABLE Matches (
    HomeTeam_ID INT UNSIGNED NOT NULL REFERENCES Teams(TeamID),
    OpponentTeam_ID INT UNSIGNED NOT NULL REFERENCES Teams(TeamID),
    Date DATE NOT NULL,
    Results VARCHAR(20) NOT NULL,
    PRIMARY KEY (HomeTeam_ID, OpponentTeam_ID)
);
